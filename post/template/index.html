isCJKLanguage: true
# 摘要信息
summary: "Hugo文档模板"
# 如果draft为true，除非使用 --buildDrafts 参数，否则不会发布文章

# 设置文章的过期时间，如果是已过期的文章则不会发布，除非使用 --buildExpired 参数

# 设置文章的发布时间，如果是未来的时间则不会发布，除非使用 --buildFuture 参数

linkTitle:

include_toc: true
